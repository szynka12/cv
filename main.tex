	\begin{cvsection}{Research interests}
		\textbf{Flows in porous materials}\par
		{\itshape\comment{%
			I am studying laminar and turbulent flow behaviour at the microscale to
			understand how pore-level dynamics impact macro scale transport in
			complex systems. The behaviour of the fluid at the porous-fluid
			interface, including the properties of turbulence, dispersion and
			porous-induced drag are the main focus of my work.
			}}\par
		\vspace{\parspace}
		\textbf{Upscaling methods}\par
		{\itshape\comment{%
			I am developing and applying upscaling methods, including Large Eddy
			Simulation and Volume Averaging Method, to improve currently used
			closures for flows in porous materials---in particular at the
			porous–fluid interface, as these two modelling strategies intersect
			there.
			}}\par

	\end{cvsection}
